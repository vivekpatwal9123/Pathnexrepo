    stage('Parallel Tasks'){
        parallel{
          stage('build') {
             steps{
               sh 'mvn clean compile'
                }
              }

           stage('Test') {
                 steps{
                  sh 'mvn test'
                 }
              }
           }    
        }

   }






stages:
 - build-test

maven-build:
  stage: build-test
  image: maven:3.8.1-jdk-17
  script:
   - git clone https://github.com/Pathnex/sample-java-app.git
   - cd sample-java-app
   - mvn clean compile
  parallel: 1

maven-test:
 stage: build-test
 image: maven:3.8.1-jdk-17
 script:
   - cd sample-java-app
   - mvn test

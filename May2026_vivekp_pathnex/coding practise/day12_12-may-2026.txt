#!/bin/bash

SOURCE="/var/log"
DEST="/backup/pathnex-$(date+%F).tar.gz"

tar -czf $DEST $SOURCE
echo "Backup created at: $DEST"



pipeline {
    agents any
    environment{
        INSTITUTE_NAME = Pathnex"

    }
    stages{
        stage('checkout') {
            steps{
                checkout scm
            }
        }
        stage('Build') {
            steps{
                sh 'mvn clean compile'

            }
        }

    }
}




stages:
 - build


build-main:
 stage: build
 image: maven:3.8.1-jdk-17
 script:
  - mvn clean compile

only:
 - main

 stage: build
 image: 3.8.1-jdk-17

  - mvn clean compile

    agent any 
    environment {
        INSTITUTE_NAME = "Pathnex"
    }
    stages {
        stage('Deploy to Dev') {
            steps {
                sh 'echo "Deploying $INSTITUTE_NAME to dev env'
            }
        }
        stage('Deploy') {
            steps {
                input message: " Approve staging Deployment?"
                sh 'echo "Deploying $INSTITUTE_NAME to staging env"'
            }
        }
        stage('Deploy to prod') {
            steps {
                input message: "Approve Prod Deployment?"
                sh 'echo "Deploying $INSTITUTE_NAME to prod env"'
            }
        }
    }
 }

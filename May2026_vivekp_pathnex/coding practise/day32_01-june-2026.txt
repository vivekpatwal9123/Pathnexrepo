---
- name: Install Nginx on Pathnex server
  hosts: all
  became: yes
  tasks: 
  - name: Install Nginx
    yum:
     name: Nginx
     state: present


    provider "aws" {
        region = "us-east-1"
    } 

    resource "aws_instance" "Pathnex-ec2" {
        ami  = "ami-xxxxxxxxx"
        instance_type = "t2.micro"

        tags = {
            Name  = "Pathnex"

        }
    }



apiVersion: apps/v1
kind: Pod
metadata:
 name: pathnex-Pod

spec:
 containers:
  - name: web
    image: nginx:latest
    ports: 
     - contaiinerPort: 80

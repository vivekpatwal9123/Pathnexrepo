---
- name: Update SSh config for Pathnex 
  hosts: all 
  become: yes

  tasks:
   - name: Disable root login 
     lineinfile:
         path: /etc/ssh/sshd_config
         regexp: "PermitRootLogin"
         line: "PermitRootlogin no"  


resource "aws_key_pair" "PathnexKey" {
    key_name  = "PathnexKey"
    public_key  = file("~/.ssh/id_rsa.pub")
}           

output "Pathnex_key_Name" {
    value = aws_key_pair.PathnexKey.key_name
}



apiVersion: v1
kind: Pod
metadata:
 name: pathnex-sidecar 
spec:
 containers:
  - name: app 
    image: nginx
  - name: log-writer 
     image: busybox
     command: ["sh", "-c", "while true; do echo log entry; sleep 5; done"]


#!/bin/bash
DIR="/var/log"

COUNT=$(ls $DIR | wc -l)

echo "Total file in $DIR: $COUNT"



pipeline {
    agent any 
    environment {
        INSTITUTE_NAME = "Pathnex"
        TEAM = "Devops"
        ENV = " staging"

    }
    stages {
        stage('Checkout') {
            steps {
                git url: 'https://github.com/pathnex/sample-java-app.git'
            }

        }
        stage('Build') {
            steps {
                sh '''
                 if [-d ".m2]; then 
                  echo Using cached dependencies 

                 else 
                  mvn clean compile 
                 fi 
                 '''   
            }
        }
    }
}



stages:
 - build 

variables:
 INSTITUTE_NAME: "Pathnex"
 TEAM: "Devops"
 ENV: "staging"

Build_stage:
  stage: build 
  image: maven:3.8.1-jdk-17
  cache:
   key: "${ENV}-maven-cache"
   paths:
    - .m2/repository
  script:
   - mvn clean compile
   - echo "Cache dependencies for $INSTITUTE_NAME in $ENV by $TEAM"

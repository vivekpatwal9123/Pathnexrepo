 - name: Install Nginx 
   hosts: all
   become: yes

   tasks:
    - name: install Nginx
      yum:
       name: nginx
       state: present

   - name: Service Start
     service:
      name: nginx
      state: started
      enabled: yes

pipeline {
    agent any
    tools {
        maven 'Maven-3.8.1'
        jdk 'JDK_17'
    }
    stages {
        stage('Checkout') {
            steps {
                git branch: "main", url: 'https://github.com/pathnex/devops'
            }
        }
        stage('Complie') {
            steps {
                sh 'mvn clean compile'
            }
        }
        stage('Test') {
            steps {
                sh 'mvn test'
            }
        }
        stage('Pakcage') {
            steps {
                sh 'mvn package'
            }
        }
    }
}



stages:
 - build
 - test
 - package

#!/bin/bash
SERVICE="sshd"

if systemctl is-actice --quiet $SERVICE; then
  echo "$SERVICE is running"

else"$SERVICE is not running"
fi



pipeline {
    agents any
    environment {
        INSTITUTE_NAME = "Pathnex"

    }
    stages{
        stage('Build') {
            steps{
                sh 'mvn clean package'
            }
        }
        stage('Promote') {
            steps{
                input message: "Promote artifacts to staging?"
                sh 'echo "Promoting artifacts for $INSTITUTE_NAME'
            }
        }
    }
}


stages:
 - build
 - promote

stage: build
image: maven:3.8.1-jdk-17
script:
 - mvn clean package

  paths:
    - target/*.jar 
  expire_in: 1 week

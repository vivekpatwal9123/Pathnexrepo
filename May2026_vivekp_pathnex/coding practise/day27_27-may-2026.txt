---
- name: Deploy Serverless lambda Functions
  hosts: localhost
  tasks:
   - name: Create Lambda Functions
     command: aws lambda create-function --function-name pathnex-lambda --runtime nodejs14.x --role arn:aws:iam::123456789:role/lambda-role --handler index.handler --zip-file fileb://lambda.zip


resource "aws_dynamodb_table" "pathnex_table" {
    name  = "pathnex-table"
    billing_mode = "PAY_PER_REQUEST"
    hash_key  = "id"
    attribute {
         name = "id"
         type = "5"
    }
}


apiVersion: kubeless.io/v1beta1
kind: Function
metadata:
 name: pathnex-lambda
 namespace: default

spec:
 runtime: nodejs14
 handler: index.handler
 function: |
  module.exports = function(event, context) {
     context.succeed('Hello World');
  }; 



  pipeline {
    agent any 
    stages {
        stage('Build') {
            steps {
                echo 'Building Lambda function package...'
                sh 'zip -r lambda.zip lambda/'
            }
        }
        stage('Deploy to Lambda') {
            steps {
                script {
                    sh 'aws lambda update-function-code --function-name pathnex-lambda --zip-file fileb://lambda.zip'

                }
            }
        }
    }
  }


docker exec -t pathnex-nginx bash
docker stats


docker run -d --memory="512m"  --cpus="1.0" nginx

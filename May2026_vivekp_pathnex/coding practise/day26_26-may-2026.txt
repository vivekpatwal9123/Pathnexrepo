---
- name: Implementing SSH-key-based Authentication
  hosts: all
  become: yes
  tasks:
   - name: Generate SSH key pair 
     command: ssh-keygen -t rsa -b 2048 -f /home/user/.ssh/id_rsa -N""
    - name: Add SSH public key to authorized_keys
      authorized_key:
        user: user
        state: present
        key: "{{ lookup('file', '/home/user/.ssh/id_rsa.pub')}}"

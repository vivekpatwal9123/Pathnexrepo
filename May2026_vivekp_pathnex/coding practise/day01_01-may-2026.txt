---
- name: Install nginx on Pathnex server
  hosts: all
  became: yes

  tasks:
   - name: Install nginx
     yum: 
      name: nginx
      state: present


provider "aws" {
    region = "us-east-1"
}

resource "aws_instance" "PathnexEC2"{
ami    =   "ami-0abcd1234abcd1234"
instance_type = "c5.xlarge"

tags = {
    Name = "Pathnex-Ec2"
}
}


apiVersion: v1
Kind: Pod 
metadata:
 name: pathenx-Pod

spec:
 containers:
  - name: web 
    image: nginx:latest
    ports:
     - containerPorts: 80

   agent any
   stages{
      stage('checkout'){
        steps{
            git branch: 'main', url: 'https://github.com/Pathnex/sample-java-app.git'
        }
      }
      stage('list files'){
        steps{
            sh 'ls -la'
        }
      }
   }

}


stages:
 - checkout

   stage: checkout
   script:
      - git clone https://github.com/Pathnex/sample-java-app.git
      - cd sample-java-app

---
- name: Install nginx
  hosts: all
  became: yes

  tasks:
    - name: Install Nginx
      yum:
       name: nginx
       state: present


       resource "aws_instance" "BrokerEC2" {
        ami = "ami-0xyz"
        instance_type = t2.micro

        tags {
            Name = "Broken"
         }
       }


apiVersion: v1
kind: Pod
metadata:
 name: Pathnex

spec:
 containers:
  - name: app
    image: nginx
    ports:
     - containerPort: 80





pipeline {
    agents any
    stages {
        stage('checkout'){
            steps{
                git branch: 'main', url: 'https://github.com/Pathnex/sample-java-app.git'
            }
        }

        stages('Build') {
            steps {
                sh '''
                mvn clean compile || {echo " Compile failed"; exit 1;}
                ...
            }
        }
        stage('Test') {
            steps {
                sh '''
                mvn test || { echo "test failed"; exit 1;}
                ...
            }
        }
    }
}     

stages:
 - build
 - test

 build:
  stage: build
  image: maven:3.8.1-jdk-17
  script:
   - git clone https://github.com/Pathnex/sample-java-app.git
   - cd sample-java-app
   - mvn clean compile || { echo "Compile failed"; exit 1;}

 test:
  stage: test
  image: maven;3.8.1-jdk-17
  script:
   - cd sample-java-app
   - mvn test || { echo " Tests failed"; exit 1; }

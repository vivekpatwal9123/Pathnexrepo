- name: Deploy Nginx across multiple kubernetes cluster
  hosts: localhost
  tasks:
   - name: Config kubectl for cluster 1
     command: kubectl config use-context  us-east-cluster

   - name: DeployNginx on cluster 1
     command: kubectl apply -f kubernetes/deployment.yaml

   - name: Configure kubectl for cluster 2
     command: kubectl config use-context us-west-cluster

   - name: Deploy Nginx on cluster 2
     command: kubectl apply -f kubernetes/deployment.yaml
